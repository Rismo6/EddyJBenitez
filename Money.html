let words= ["Succeed", "Persevere", "Strong-willed"];
let word= "Money.";

function setup() {
createCanvas(1200, 400);

}

function draw() {
if (frameCount === 1){
capturer.start();

}

background(0,170,170);

//Shaking text. My y For loop [words]
for(j=25; j<=400; j+=25){
let r= random(40,50);

//List "Succeed", "Persevere", "Strong-willed"
textSize(20);
fill(0,0,0);
text(words,r,j);

//black bar
fill(0,0,0);
rect(0,310,400,30);

//lifted text [words]
fill(150,0,0);
textSize(23);
text(words,r,330);

//fixed text. My y for loop 'Money'
for(m=25; m<=340; m+=45){

//dropping text 'Money'
textSize(250);
fill(150,0,0);
text(word, 410,m);

//overlap the for loop
fill(0,170,170);
noStroke();
rect(400,220,800,150);

//fixed text
fill(0,0,0);
text(word, 410,340);

}
}

if(frameCount <1000){
capturer.capture(canvas)
}else if(frameCount === 1000){
capturer.stop();
capturer.save();

}
}
