let angle = 0;
var x = 200;
var y = 200;


function setup() {
createCanvas(1200, 400);
angleMode(DEGREES);
}

function draw() {
if(frameCount == 1) {
capturer.start();
}
background(220);

//w1
push();
translate(x, y);
circle()
strokeWeight(4);
fill(237,41,57);
rect(-199,-200,400,400);

//w1 white circle
noFill();
strokeWeight(2);
rotate(angle);
stroke(250,250,250);
circle(0,0,100);

//w1 black outline
noFill();
stroke(0,0,0);
circle(0,0,105);
circle(0,0,95);

//w1 in curves
stroke(0,0,0);
rotate(angle);
curve(0, 250, -40, -10, 40, -10, 0, 250);

curve(0, -250, -40, 10, 40, 10, 0, -250);

//w1 arms
stroke(0,0,0);
fill(255,255,255);
curve(-400,0,50.5,10,50.5,-10,-400,0);

curve(400,0,-50.5,10,-50.5,-10,400,0);
angle = angle + .5;
pop();

//b2
push();
translate(x * 3, y);
strokeWeight(4);
fill(213,159,0);
rect(-199,-200,799,400);

// b2 white circle
noFill();
strokeWeight(2);
rotate(angle);
stroke(96,4,00);
circle(0,0,100);

//b2 black outline
noFill();
stroke(0,0,0);
circle(0,0,105);
circle(0,0,95);

//b2 in curves
stroke(0,0,0);
rotate(angle);
curve(0, 250, -40, -10, 40, -10, 0, 250);

curve(0, -250, -40, 10, 40, 10, 0, -250);

//b2 arms
stroke(0,0,0);
fill(96,4,00);
curve(-400,0,50.5,10,50.5,-10,-400,0);

curve(400,0,-50.5,10,-50.5,-10,400,0);

pop();

//b3
push();
translate(x * 5, y);

noFill();
strokeWeight(2);

//b3 white circle
rotate(angle);
stroke(60,5,00);
circle(0,0,100);

//b3 black outline
noFill();
stroke(0,0,0);
circle(0,0,105);
circle(0,0,95);

//b3 in curves
stroke(0,0,0);
rotate(angle);
curve(0, 250, -40, -10, 40, -10, 0, 250);

curve(0, -250, -40, 10, 40, 10, 0, -250);

//b3 arms
stroke(0,0,0);
fill(40,5,0);
curve(-400,0,50.5,10,50.5,-10,-400,0);

curve(400,0,-50.5,10,-50.5,-10,400,0);

pop();

if(frameCount < 1000){
capturer.capture(canvas)
} else if (frameCount == 1000){

capturer.save()
capturer.stop()
}
}
